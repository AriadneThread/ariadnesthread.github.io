#处理页面中图片变形的问题


适应的情况：比如页面中图片的固定大小为200*100，但是后台传过来的图片数据大小是300*50 300*200, 100*200,100*50等等的情况，处理的原理就是，首先要占满目标图片，根据判断，先把高度或者宽度设置为目标图片的大小，剩下另一边多出来的用margin，把图片居中显示就可，具体代码如下：

```
    function DrawImage(ImgD,src,w,h){
        var image=new Image();
        image.src=src;       
        var width=(h/image.height)*image.width;//固定高度，计算宽度
        var height=(w/image.width)*image.height;//固定宽度,计算高度      
        //alert(image.width+"="+image.height+"######"+width+"=定高"+h+"######定高"+w+"="+height);
        if((width-w)>0&&(height-h)>0){
            if((width-w)>(height-h)){
                var img_w=width+"px";
                var img_h=h+"px";
                var img_left="-"+(width-w)/2+"px";
                $(ImgD).css({width:img_w,height:img_h,marginLeft:img_left});
            }else{  //相反
                var img_w=w+"px";
                var img_h=height+"px";
                var img_top="-"+(height-h)/2+"px";
                $(ImgD).css({width:img_w,height:img_h,marginTop:img_top});
            }
        }     
        if((width-w)<0&&(height-h)<0){
            if(Math.abs(width-w)>Math.abs(height-h)){
                var img_w=w+"px";
                var img_h=height+"px";
                var img_top="-"+(height-h)/2+"px";
                $(ImgD).css({width:img_w,height:img_h,marginTop:img_top});
            }else{
                var img_w=width+"px";
                var img_h=h+"px";
                var img_left="-"+(width-w)/2+"px";
                $(ImgD).css({width:img_w,height:img_h,marginLeft:img_left});
            }
        }
        if((width-w)>0&&(height-h)<0){
            var img_w=width+"px";
            var img_h=h+"px";
            var img_left="-"+(width-w)/2+"px";
            $(ImgD).css({width:img_w,height:img_h,marginLeft:img_left});
        }

        if((width-w)<0&&(height-h)>0){
            var img_w=w+"px";
            var img_h=height+"px";
            var img_top="-"+(height-h)/2+"px";
            $(ImgD).css({width:img_w,height:img_h,marginTop:img_top});
        }
    }
```
