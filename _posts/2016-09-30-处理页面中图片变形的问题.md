## 处理页面中图片变形的问题

适应的情况：比如页面中图片的固定大小为200x100，但是后台传过来的图片数据大小是300x50、300x200、100x200、100x50等等的情况，处理的原理就是，首先要占满目标图片，根据判断，先把高度或者宽度设置为目标图片的大小，剩下另一边多出来的用margin，把图片居中显示即可，具体代码如下：

```javascript
function DrawImage(ImgD, src, w, h) {
    var image = new Image();
    image.src = src;
    var width = (h / image.height) * image.width; //固定高度，计算宽度
    var height = (w / image.width) * image.height; //固定宽度，计算高度
    //alert("图片的宽：" + image.width + "，图片的高：" + image.height);
    //alert("宽度：" + h + "=" + width + "，高度：" + w + "=" + height);
    if ((width - w) > 0 && (height - h) > 0) {
        if ((width - w) > (height - h)) {
            var img_w = width + "px";
            var img_h = h + "px";
            var img_left = "-" + (width - w) / 2 + "px";
            $(ImgD).css({
                width: img_w,
                height: img_h,
                marginLeft: img_left
            });
        } else { //相反
            var img_w = w + "px";
            var img_h = height + "px";
            var img_top = "-" + (height - h) / 2 + "px";
            $(ImgD).css({
                width: img_w,
                height: img_h,
                marginTop: img_top
            });
        }
    }
    if ((width - w) < 0 && (height - h) < 0) {
        if (Math.abs(width - w) > Math.abs(height - h)) {
            var img_w = w + "px";
            var img_h = height + "px";
            var img_top = "-" + (height - h) / 2 + "px";
            $(ImgD).css({
                width: img_w,
                height: img_h,
                marginTop: img_top
            });
        } else {
            var img_w = width + "px";
            var img_h = h + "px";
            var img_left = "-" + (width - w) / 2 + "px";
            $(ImgD).css({
                width: img_w,
                height: img_h,
                marginLeft: img_left
            });
        }
    }
    if ((width - w) > 0 && (height - h) < 0) {
        var img_w = width + "px";
        var img_h = h + "px";
        var img_left = "-" + (width - w) / 2 + "px";
        $(ImgD).css({
            width: img_w,
            height: img_h,
            marginLeft: img_left
        });
    }
    if ((width - w) < 0 && (height - h) > 0) {
        var img_w = w + "px";
        var img_h = height + "px";
        var img_top = "-" + (height - h) / 2 + "px";
        $(ImgD).css({
            width: img_w,
            height: img_h,
            marginTop: img_top
        });
    }
}
```
